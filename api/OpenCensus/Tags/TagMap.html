Overview

TagMap

Collection of tag key and value.

Examples:


tag_map = OpenCensus::Tags::OpenCensus.new

# Add or update
tag_map["key1"] = "value1"
tag_map["key2"] = "value2"

# Get value
tag_map["key1"] # value1

# Delete
tag_map.delete "key1"

# Iterate
tag_map.each do |key, value|
  p key
  p value
end

# Length
tag_map.length # 1

Create tag map from hash


tag_map = OpenCensus::Tags::OpenCensus.new({ "key1" => "value1"})
